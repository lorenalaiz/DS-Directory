const { BasePage } = require('./BasePage');

class HomePage extends BasePage {
  /**
   * @param {import('@playwright/test').Page} page
   */
  constructor(page) {
    super(page);
    this.heading = page.locator('h1');

    this.professionalsTab = page.getByRole('button', { name: 'Professionals' });
    this.productsTab = page.getByRole('button', { name: 'Products & Toys' });
    this.familiesTab = page.getByRole('button', { name: 'Families' });
    this.marketplaceTab = page.getByRole('button', { name: 'Marketplace' });
    this.messagesTab = page.getByRole('button', { name: 'Messages' });
    this.activityTab = page.getByRole('button', { name: 'Activity' });
    this.requestsTab = page.getByRole('button', { name: 'Requests' });
    this.ndisGuideTab = page.getByRole('button', { name: 'NDIS Guide' });

    this.contactButton = page.getByRole('button', { name: 'Contact' });
    this.installAppButton = page.getByRole('button', { name: 'Install App' });
    this.adminButton = page.getByRole('button', { name: '🔐 Admin' });
    this.adminToolsButton = page.getByRole('button', { name: '⚙️ Admin tools' });
    this.backupButton = page.getByRole('button', { name: '⬇️ Backup' });
    this.restoreButton = page.getByRole('button', { name: '⬆️ Restore' });
    this.signOutButton = page.getByRole('button', { name: 'Sign out' });

    this.adminEmailInput = page.locator('#admin-email');
    this.adminPasswordInput = page.locator('#admin-password');
    this.adminSignInButton = page.getByRole('button', { name: 'Sign in' });
    this.adminLoginCancelButton = page.getByRole('button', { name: 'Cancel' });
    this.adminSignedInToast = page.getByText('Signed in as admin.');
    this.adminLoginError = page.locator('#admin-login-error');
  }

  async goto() {
    await super.goto('/');
  }

  async loginAsAdmin(email, password) {
    await this.adminButton.click();
    await this.adminEmailInput.fill(email);
    await this.adminPasswordInput.fill(password);
    await this.adminSignInButton.click();
    await this.adminSignedInToast.waitFor();
  }

  async attemptAdminLogin(email, password) {
    await this.adminButton.click();
    await this.adminEmailInput.fill(email);
    await this.adminPasswordInput.fill(password);
    await this.adminSignInButton.click();
  }

  async signOut() {
    await this.adminToolsButton.click();
    await this.signOutButton.click();
  }
}

module.exports = { HomePage };
